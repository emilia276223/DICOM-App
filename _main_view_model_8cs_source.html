243 internal void OpenImage(string file)
244 {
245 // zapisujemy pozycje centrum wycinka powększającego
246 SaveMagnifiedImageCenterPosition();
247
248 var result = LoadAndRegisterImage(file);
249 // załadowanie obrazu (jeśli się da)
250 if (result == null)
251 {
252 return; // jeśli nie udało się załadować nic więcej nie robimy
253 }
254
255 // update wykresu jeśli został zmieniony pacjent
256 if(DicomMetadata != null && result.Metadata.PatientID != DicomMetadata.PatientID )
257 ChartVM.UpdateChart(_dbHandler.GetOpticNerveLenValuesForStudiesOfPatient(result.Metadata.PatientID));
258
259 UpdateLocalState(result);
260
261 // update skali linii
262 var scaler = DICOMScalerFactory.GetDICOMScaler(file);
263 LinesB.UpdateScalerForOriginalLines(scaler, ImageVM._image_scale, ImageVM.ImageWidth, ImageVM.ImageHeight);
264
265 // Update predykcji modelu
266 UpdatePrediction(result.Image);
267
268 // okno powiększające
269 MagnifiedImageVM.UpdateImage(result.Image, ImageVM._image_scale, scaler);
270
271 }

280 private void UpdatePrediction(ImageData image)
281 {
282 if (!IsAIEnabled)
283 {
284 var p = new Avalonia.Point(0, 0);
285 LinesB.SetLinesToPrediction([p, p, p, p]);
286 return;
287 }
288
289 // predykcja punktów
290 var pred = _MLPlugin.GetPredictionFromImageData(image);
291 var scaledPred = new Avalonia.Point[4];
292 for (int i = 0; i < 4; i++) scaledPred[i] = (pred.Points[i] / ImageVM._image_scale);
293 LinesB.SetLinesToPrediction(scaledPred);
294
295 // heatmapy
296 HeatmapHandler.UpdateHeatmaps(pred.Heatmaps);
297 }
