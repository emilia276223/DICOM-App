59 private static DICOMMetadata? GetImageMetadata(DicomFile dicomFile)
60 {
61 try
62 {
63 var patientName = dicomFile.Dataset.GetSingleValueOrDefault(DicomTag.PatientName, "Unknown");
64 var savedName = patientName;
65 if (patientName.Contains("^"))
66 savedName = patientName.Split('^')[0] + " " + patientName.Split('^')[1];
67
68 var studyDate = dicomFile.Dataset.GetSingleValueOrDefault(DicomTag.StudyDate, "Unknown");
69 var savedDate = studyDate;
70 if(!studyDate.Contains("."))
71 savedDate = studyDate.Substring(0, 4) + "." + studyDate.Substring(4, 2) + "." + studyDate.Substring(6, 2);
72
73 var metadata = new DICOMMetadata
74 {
75 PatientID = dicomFile.Dataset.GetSingleValueOrDefault(DicomTag.PatientID, "Unknown"),
76 PatientName = savedName,
77 StudyDate = savedDate,
78 StudyUID = dicomFile.Dataset.GetSingleValueOrDefault(DicomTag.StudyInstanceUID, "Unknown"),
79 UID = dicomFile.Dataset.GetSingleValueOrDefault(DicomTag.SOPInstanceUID, "Unknown"),
80 };
81
82 return metadata;
83 }
84 catch { return null; }
85 }

94 private static (int[], int, int) CropWithAddingPadding(int[] original, int originalWidth, int originalHeight)
95 {
96 int width = 560;
97 int height = 680;
98 int[] res = new int[width * height];
99 for (int y = 0; y < height; y++)
100 {
101 for (int x = 0; x < width; x++)
102 {
103 // jeśli poza obrazem to ustawiamy na 0
104 if (x >= originalWidth || y >= originalHeight)
105 {
106 res[y * width + x] = 0;
107 }
108
109 // kopiujemy piksel z oryginału do nowej tablicy
110 else
111 {
112 res[y * width + x] = original[y * originalWidth + x];
113 }
114 }
115 }
116 return (res, width, height);
117 }

126 private static (int[], int, int) CropPixels(int[] original, int originalWidth, int originalHeight)
127 {
128 // jeśli obraz ma inne wymiary niż przewidziane to przycinamy tak, żeby się zgadzało wielkością
129 if (originalWidth < 890 || originalHeight < 735)
130 {
131 return CropWithAddingPadding(original, originalWidth, originalHeight);
132 }
133
134 // chcemy przyciąć do samego obrazu (bez marginesów)
135 // interesują nas piksele: w pionie od 55 do 735, w poziomie od 330 do 890
136 var startX = 330;
137 var endX = 890;
138 var startY = 55;
139 var endY = 735;
140
141 int width = endX - startX; // 560
142 int height = endY - startY; // 680
143 int[] cropped = new int[width * height];
144
145
146 // kopiujemy piksele z oryginału do nowej tablicy
147 for (int y = startY; y < endY; y++)
148 {
149 for (int x = startX; x < endX; x++)
150 {
151 cropped[(y - startY) * width + (x - startX)] = original[y * originalWidth + x];
152 }
153 }
154
155 return (cropped, width, height);
156 }

165 private static ImageData? GetImage(DicomFile dicomFile)
166 {
167 try {
168 var dicomImage = new FellowOakDicom.Imaging.DicomImage(dicomFile.Dataset);
169 var rendered = dicomImage.RenderImage();
170
171
172 int[] src = rendered.Pixels.Data;
173
174 int originalWidth = rendered.Width;
175 int originalHeight = rendered.Height;
176
177 int[] cropped;
178 int newWidth, newHeight;
179
180
181 (cropped, newWidth, newHeight) = CropPixels(src, originalWidth, originalHeight);
182 return new ImageData(newWidth, newHeight, cropped);
183 }
184 catch { return null; }
185 }
