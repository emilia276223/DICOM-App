31 {
32 anonimizedData = _dbHandler.GetAnonimizedData();
33 magnifiedCentersData = _dbHandler.GetMagnifiedCentersData();
34 }
35 catch { return false; }
36
37 // zrobienie słownika tak, by id obrazów zostały przetłumaczone na oryginalne wartości
38 var imagesWithInformationUID = new Dictionary<string, string>();
39 for (var i = 0; i < anonimizedData.Count; i++)
40 {
41 var id = anonimizedData[i][0];
42 if (!imagesWithInformationUID.ContainsKey(id))
43 {
44 imagesWithInformationUID.Add(id, Guid.NewGuid().ToString());
45 }
46 anonimizedData[i][0] = (imagesWithInformationUID[id]);
47 }
48 for (var i = 0; i < magnifiedCentersData.Count; i++)
49 {
50 var id = magnifiedCentersData[i][0];
51 if (!imagesWithInformationUID.ContainsKey(id))
52 {
53 imagesWithInformationUID.Add(id, Guid.NewGuid().ToString());
54 }
55 magnifiedCentersData[i][0] = (imagesWithInformationUID[id]);
56 }
57
58 // wyeksportowanie kopii plików PNG reprezentujących obrazy, których dane zostały wyeksportowane
59 foreach (var image in imagesWithInformationUID)
60 {
61 PNGFileStorageService.ExportSavedPNGImage(image.Key, image.Value);
62 }
63
64 // zapisanie do pliku
65 _csvHandler.WriteExportedImageData(anonimizedData);
66 _csvHandler.WriteMagnificationData(magnifiedCentersData);
67
68 // ZIP wyeksportowanych danych
69 var sourcePath = AppDirectoryHandler.Instance.GetExportedDataDirectory();
70 var zipPath = AppDirectoryHandler.Instance.GetExportedZIPFilename();
71 if (System.IO.File.Exists(zipPath))
72 {
73 System.IO.File.Delete(zipPath);
74 }
75 System.IO.Compression.ZipFile.CreateFromDirectory(sourcePath, zipPath);
76
77 return true;
78 }
