37 private List<string> GetAllFilesFromDirectory(string dir)
38 {
39 var files = new List<string>();

41 // nie powinno się wydarzyć, ale na wszelki wypadek można sprawdzić
42 if (!Directory.Exists(dir))

59 private async Task<List<string>> PickFolderAsync(string? startPath)
60 {
61 if (startPath == null)
62 {
63 startPath = Environment.GetFolderPath(Environment.SpecialFolder.UserProfile);
64 }
65
66 var options = new FolderPickerOpenOptions
67 {
68 Title = "Select DICOM folder",
69 AllowMultiple = false,
70 SuggestedStartLocation = await _sp.TryGetFolderFromPathAsync(startPath),
71 };
72
73 var dir = await _sp.OpenFolderPickerAsync(options);
74
75 // jeśli nie wyszło zwracamy pustą listę
76 if(dir == null)
77 {
78 return new List<string>();
79 }
80
81 // zwracamy wszystkie pliki z wybranego katalogu
82 var dirPath = dir.FirstOrDefault()?.Path.LocalPath;
83 return GetAllFilesFromDirectory(dirPath);
84 }

93 private async Task<string?> PickFileAsync(string? startPath, string fileType, List<string> patterns)

95 if (startPath == null)

99
100 var options = new FilePickerOpenOptions
101 {
102 Title = "Select a file",
103 AllowMultiple = false,
104 SuggestedStartLocation = await _sp.TryGetFolderFromPathAsync(startPath),
105 FileTypeFilter =
106 [

109 Patterns = patterns

111

113 };
114
115 var files = await _sp.OpenFilePickerAsync(options);
