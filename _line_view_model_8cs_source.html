112 private void UpdateCaps()
113 {
114 double dx = StartPoint.X - EndingPoint.X;
115 double dy = StartPoint.Y - EndingPoint.Y;
116 double len = Math.Sqrt(dx * dx + dy * dy);
117
118 // jeśli linia jest zbyt krótka nie ma bzegów
119 if (Math.Abs(dx) < 3 && Math.Abs(dy) < 3) {
120 CapStart1 = StartPoint;
121 CapStart2 = StartPoint;
122 CapEnd1 = EndingPoint;
123 CapEnd2 = EndingPoint;
124 return;
125 }
126
127 // jeśli linia jest inna to brzegi ustawiamy pod kątem prostym do linii
128 double sin = _capLength * dx / len;
129 double cos = _capLength * dy / len;
130 CapStart1 = new Point(StartPoint.X - cos, StartPoint.Y + sin);
131 CapStart2 = new Point(StartPoint.X + cos, StartPoint.Y - sin);
132 CapEnd1 = new Point(EndingPoint.X - cos, EndingPoint.Y + sin);
133 CapEnd2 = new Point(EndingPoint.X + cos, EndingPoint.Y - sin);
134 }
