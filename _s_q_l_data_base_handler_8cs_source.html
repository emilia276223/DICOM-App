119 public List<(string,double)> GetOpticNerveLenValuesForStudiesOfPatient(string patientID)
120 {
121 try
122 {
123 if (_conn == null) return new List<(string, double)>();
124 _conn.Open();
125 using (var cmd = _conn.CreateCommand())
126 {
127 cmd.CommandText =
128 """
129 SELECT study_id, studies.date, MAX(optic_nerve_size)
130 FROM images
131 JOIN studies ON studies.id = images.study_id
132 WHERE optic_nerve_size IS NOT NULL
133 AND studies.patient_id = @pid
134 GROUP BY study_id
135 """;
136 cmd.Parameters.AddWithValue("pid", patientID);
137
138
139 var res = new List<(string, double)>();
140 using (var resReader = cmd.ExecuteReader())
141 {
142 while (resReader.Read())
143 {
144 var studyId = resReader.GetString(0);
145 var studyDate = resReader.GetString(1);
146 var maxSize = resReader.GetDouble(2);
147
148 res.Add((studyDate, maxSize));
149 }
150 }
151 _conn.Close();
152 return res;
153 }
154 }
155 catch (SqliteException)
156 {
157 return new List<(string, double)>();
158 }
159 }

162 public List<List<string>> GetAnonimizedData()
163 {
164 if (_conn == null) return new List<List<string>>();
165 _conn.Open();
166 var res = new List<List<string>>();
167 using (var cmd = _conn.CreateCommand())
168 {
169 // wybieramy tylko obrazy z pełnymi informacjami
170 cmd.CommandText =
171 """
172 SELECT id, optic_nerve_size, d1x, d1y, d2x, d2y, d3x, d3y, d4x, d4y
173 FROM images
174 WHERE optic_nerve_size IS NOT NULL
175 AND d1x IS NOT NULL
176 AND d2x IS NOT NULL
177 AND d3x IS NOT NULL
178 AND d4x IS NOT NULL
179 AND d1y IS NOT NULL
180 AND d2y IS NOT NULL
181 AND d3y IS NOT NULL
182 AND d4y IS NOT NULL
183 """;
184
185 // stworzenie linii do zapisania do csv
186 using (var resReader = cmd.ExecuteReader())
187 {
188 while (resReader.Read())
189 {
190 // jeśli są to wartości to dodajemy je do listy
191 List<string> data = [
192 resReader.GetString(0), // imageID
193 resReader.GetDouble(1).ToString(CultureInfo.InvariantCulture) // opticNerveSize
194 ];
195
196 for (int i = 2; i <= 9; i++)
197 {
198 data.Add(
199 resReader.GetDouble(i).ToString(CultureInfo.InvariantCulture)); // points
200 }
201 res.Add(data);
202 }
203 }
204 }
205 _conn.Close();
206 return res;
207 }
